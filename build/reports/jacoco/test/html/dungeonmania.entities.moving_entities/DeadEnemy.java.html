package dungeonmania.entities.moving_entities;

import java.util.Arrays;
import java.util.List;

import dungeonmania.dungeon.Dungeon;
import dungeonmania.entities.Entity;
import dungeonmania.response.models.AnimationQueue;
import dungeonmania.util.Direction;
import dungeonmania.util.Position;

public class DeadEnemy extends MovingEntity {
    private String type;

    public DeadEnemy(Entity entity, Position position) {
        super("none", position);
        type = entity.getType();
    }
    @Override
    public void onTick() {
        Dungeon.getInstance().getEntities().remove(this);     
    }

    @Override
    public boolean onOverlap(Entity entity, Direction dir) {
        // TODO Auto-generated method stub
        return false;
    }

    @Override
    public void init() {
        // TODO Auto-generated method stub
        
    }

    @Override
    public List<AnimationQueue> getAnimationResponse() {
        if (type == "spider") {
            animationBuffer.add(new AnimationQueue("PostTick", getId(), Arrays.asList("translate-x 1, over 0.5s", "tint 0xff0000", "sprite spider"), false, -1));
        }
        if (type == "zombie_toast") {
            animationBuffer.add(new AnimationQueue("PostTick", getId(), Arrays.asList("translate-x 1, over 0.5s", "tint 0xff0000", "sprite zombie_toast"), false, -1));
        }
        if (type == "mercenary") {
            animationBuffer.add(new AnimationQueue("PostTick", getId(), Arrays.asList("translate-x 1, over 0.5s", "tint 0xff0000", "sprite mercenary"), false, -1));
        }
        if (type == "assassin") {
            animationBuffer.add(new AnimationQueue("PostTick", getId(), Arrays.asList("translate-x 1, over 0.5s", "tint 0xff0000", "sprite assassin"), false, -1));
        }
        if (type == "hydra") {
            animationBuffer.add(new AnimationQueue("PostTick", getId(), Arrays.asList("translate-x 1, over 0.5s", "tint 0xff0000", "sprite hydra"), false, -1));
        }
        return super.getAnimationResponse();
    }
    
}
